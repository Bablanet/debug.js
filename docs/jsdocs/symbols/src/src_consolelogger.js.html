 49 ConsoleLogger.formatterIdentity	= function(args, stackFrame, severity){
 50 	return args;
 51 }

 53 ConsoleLogger.formatterTimeStamp	= function(args, stackFrame, severity){
 54 	// build prefix with time
 55 	var present	= new Date();
 56 	var prefixColor	= ConsoleLogger._formatterSeverityColor(severity);
 57 	var prefix	= prefixColor + pad(present.getHours(),2)
 58 				+ ':'
 59 				+ pad(present.getMinutes(),2)
 60 				+ ':'
 61 				+ pad(present.getSeconds(),2)
 62 				+ ConsoleLogger._formatterColor.reset;
 63 	// convert arguments into actual Array
 64 	args	= Array.prototype.slice.call(args, 0);
 65 	// prepend the prefix
 66 	args.unshift(prefix);
 67 	// return the result
 68 	return args;
 69 
 70 	function pad(val, len) {
 71 		val = String(val);
 72 		while(val.length < len) val = "0" + val;
 73 		return val;
 74 	};
 75 };

 77 ConsoleLogger.formatterOrigin	= function(args, stackFrame, severity)
 78 {
 79 	// compute prefix
 80 	var prefixColor	= ConsoleLogger._formatterSeverityColor(severity);
 81 	var prefix	= prefixColor + stackFrame.originId() + ConsoleLogger._formatterColor.reset;
 82 	// convert arguments into actual Array
 83 	args		= Array.prototype.slice.call(args, 0);
 84 	// prepend the prefix
 85 	args.unshift(prefix);
 86 	// return the result
 87 	return args;
 88 }

103 /**
104  * return ainsi color per intensity
105  * @param  {String} severity the severity of the message
106  * @return {String} ansi color string
107  */
108 ConsoleLogger._formatterSeverityColor	= function(severity){
109 	var color	= ConsoleLogger._formatterColor;
110 	if( severity === 'log' )	return color.red;
111 	if( severity === 'warn' )	return color.purple;
112 	return color.green;
113 }

124 /**
125  * Store all the filters
126  * @type {Array}
127  */
128 ConsoleLogger._filters	= [];
129 
130 /**
131  * push a new filter - stackFrame 
132  * 
133  * @param  {Function(stackFrame)} validFor function which determine which stackFrame is valid 
134  * @param  {[type]} severity [description]
135  * @return {[type]}          [description]
136  */
137 ConsoleLogger.pushFilter	= function(validFor, severity){
138 	// handle polymorphism
139 	if( typeof(validFor) === 'string' ){
140 		var path	= validFor;
141 		return ConsoleLogger.pushFilter(function(stackFrame, severity){
142 			return stackFrame.url.lastIndexOf(path) === stackFrame.url.length - path.length ? true : false;
143 		}, severity);
144 	}else if( validFor instanceof RegExp ){
145 		var regexp	= validFor;
146 		return ConsoleLogger.pushFilter(function(stackFrame, severity){
147 			return stackFrame.url.match(regexp)	? true : false
148 		}, severity);		
149 	}
150 	// sanity check - sanity level MUST be defined
151 	console.assert(Object.keys(ConsoleLogger.Severity).indexOf(severity) !== -1, 'unknown severity level');
152 	console.assert(validFor instanceof Function);
153 	// push new level
154 	ConsoleLogger._filters.push({
155 		validFor	: validFor,
156 		severity	: ConsoleLogger.Severity[severity]
157 	});
158 };
159 
160 /**
161  * test if a given instance of severity and stackframe is valid for ConsoleLogger
162  * @param  {string} severity   Severity to test
163  * @param  {Object} stackFrame stackframe to test (in stacktrace.js format)
164  * @return {Boolean}           true if the filter is valid, false otherwise
165  */
166 ConsoleLogger.filter	= function(severity, stackFrame){
167 	stackFrame	= stackFrame	|| Stacktrace.parse()[2];
168 	// sanity check - sanity level MUST be defined
169 	console.assert(Object.keys(ConsoleLogger.Severity).indexOf(severity) !== -1, 'unknown severity level');
170 	// go thru all the filters
171 	var filters	= ConsoleLogger._filters;
172 	for(var i = 0; i < filters.length; i++){
173 		var filter	= filters[i];
174 		if( filter.validFor(stackFrame) === false )	continue;
175 		return ConsoleLogger.Severity[severity] >= filter.severity ? true : false;
176 	}
177 	// if this point is reach, check with Severity.dfl
178 	return ConsoleLogger.Severity[severity] >= ConsoleLogger.Severity.dfl ? true : false;
179 }

185 ConsoleLogger._print	= function(severity, args){
186 	var stackFrame	= Stacktrace.parse()[2];
187 
188 	if( ConsoleLogger.filter(severity, stackFrame) === false )	return;
189 
190 	var args	= ConsoleLogger.formatter(args, stackFrame, severity);
191 	var _console	= ConsoleLogger._origConsole;
192 	_console[severity].apply(_console.console, args);	
193 };
