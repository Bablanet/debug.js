1 /**
  2  * @namespace logger compatible with console.* calls
  3  */
  4 var ConsoleLogger	= {};
  5 
  6 var parseStacktrace	= parseStacktrace	|| require('./parsestacktrace.js');
  7 
  8 ConsoleLogger.Severity	= {
  9 	'none'	: 0,
 10 	'log'	: 1,
 11 	'warn'	: 2,
 12 	'error'	: 3
 13 };
 14 
 15 ConsoleLogger.Severity.dflLevel	= ConsoleLogger.Severity.warn;
 16 
 17 ConsoleLogger._filter	= function(stackFrame, severity){
 18 	var stacktrace	= parseStacktrace();
 19 	var stackFrame	= stacktrace[2];
 20 	console.log('url', stackFrame.url)
 21 	//var keepIt	= ConsoleLogger._filter(stackFrame, ConsoleLogger.SEVERITY_LOG)
 22 	
 23 	return true;
 24 }
 25 
 26 ConsoleLogger.log	= function(/* ... */){
 27 	if( ConsoleLogger._filter('log') === false )	return;
 28 	console.log.apply(console, arguments);
 29 }
 30 
 31 ConsoleLogger.warn	= function(/* ... */){
 32 	if( ConsoleLogger._filter('warn') === false )	return;
 33 	console.warn.apply(console, arguments);	
 34 }
 35 
 36 ConsoleLogger.error	= function(/* ... */){
 37 	if( ConsoleLogger._filter('error') === false )	return;
 38 	console.error.apply(console, arguments);	
 39 }
 40 
 41 // export the class in node.js - if running in node.js
 42 if( typeof(window) === 'undefined' )	module.exports	= ConsoleLogger;
