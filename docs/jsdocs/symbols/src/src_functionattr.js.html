 15 /**
 16  * Function attribute creator
 17  * 
 18  * @return {FnAttrClass} a FnAttrClass builder
 19 */
 20 var fnAttr	= function(originalFn, fnName){
 21 	return new FnAttrClass(originalFn, fnName)
 22 }
 23 
 24 /**
 25  * Wrap a function between 2 functions
 26  * 
 27  * @param {Function} originalFn the original function
 28  * @param {Function} beforeFn the function to call *before* the original function
 29  * @param {Function} afterFn the function to call *after* the original function
 30  * @returns {Function} The modified function
 31 */
 32 fnAttr.wrapCall	= function(originalFn, beforeFn, afterFn){
 33 	return function(){
 34 		var stopNow	= false;
 35 		// call beforeFn if needed
 36 		if( beforeFn )	stopNow = beforeFn(originalFn, arguments);
 37 		// forward the call to the original function
 38 		var result	= originalFn.apply(this, arguments);
 39 		// call afterFn if needed
 40 		if( afterFn )	afterFn(originalFn, arguments, result);
 41 		// return the result
 42 		return result;
 43 	}
 44 }

 50 /**
 51  * Internal class to build the attributes on the funciton
 52  * 
 53  * @constructor
 54  * 
 55  * @param {Function} originalFn the function on which the attributes are set
 56  * @param {String}   fnName     optional name of the function - default to 'aFunction'
 57  */
 58 var FnAttrClass	= function(originalFn, fnName){
 59 	this._currentFn	= originalFn;
 60 	this._fnName	= fnName	|| 'aFunction'
 61 }
 62 
 63 /**
 64  * mark the end of the attributes declaration
 65  * 
 66  * @return {Function} The actual function with the attributes
 67 */
 68 FnAttrClass.prototype.end	= function(){
 69 	return this._currentFn;
 70 }
 71 
 72 // export the class in node.js - if running in node.js
 73 if( typeof(window) === 'undefined' )	module.exports	= fnAttr;

 80 /**
 81  * mark the function as deprecated - aka you can use it but it will disapears soon
 82  * @param  {string} message the optional message to provide
 83  */
 84 FnAttrClass.prototype.deprecated	= function(message){
 85 	var used	= false;
 86 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
 87 		if( used )	return;
 88 		used	= true;
 89 		console.warn(message || "Deprecated function "+this._fnName+" called. Please update your code.");
 90 	}.bind(this));
 91 	return this;	// for chained API
 92 }
 93 
 94 /**
 95  * mark the function as obsolete
 96  * @param  {string} message obsolete message to display
 97  */
 98 FnAttrClass.prototype.obsolete	= function(message){
 99 	var used	= false;
100 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
101 		if( used )	return;
102 		used	= true;
103 		console.assert(false, message || "Obsoleted function "+this._fnName+" called. Please update your code.");
104 	}.bind(this));
105 	return this;	// for chained API
106 }

108 /**
109  * display a message with a timestamp every time the function is used
110  * @return {string} message optional message to display
111  */
112 FnAttrClass.prototype.timestamp	= function(message){
113 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
114 		console.log(''+ new Date + ': '+this._fnName+' being called');
115 	}.bind(this));
116 	return this;	// for chained API
117 };
118 
119 /**
120  * log a message when the function is call
121  * @param  {string} message the message to display
122  */
123 FnAttrClass.prototype.log		= function(message){
124 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
125 		console.log(message);
126 	});
127 	return this;	// for chained API
128 };

134 /**
135  * hook a function be be caller before the actual function
136  * @param  {Function} beforeFn the function to call
137  */
138 FnAttrClass.prototype.before	= function(beforeFn){
139 	this._currentFn	= fnAttr.wrapCall(this._currentFn, beforeFn, null);
140 	return this;	// for chained API
141 };
142 
143 /**
144  * hook a function to be called after the actual function
145  * @param  {Function} afterFn the function to be called after
146  */
147 FnAttrClass.prototype.after	= function(afterFn){
148 	this._currentFn	= fnAttr.wrapCall(this._currentFn, null, afterFn);
149 	return this;	// for chained API
150 };

156 /**
157  * Warp the function call in a console.time()
158  * 
159  * @param  {String} label the label to use for console.time(label)
160  */
161 FnAttrClass.prototype.time	= function(label){
162 	label	= label !== undefined ? label : this._fnName+".time()-"+Math.floor(Math.random()*9999).toString(36);
163 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
164 		console.time(label)
165 	}, function(){
166 		console.timeEnd(label)
167 	});
168 	return this;	// for chained API
169 };
