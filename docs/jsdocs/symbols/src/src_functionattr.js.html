 14 /**
 15  * Function attribute builder
 16 */
 17 var fnAttr	= function(originalFn, fnName){
 18 	return new FnAttrClass(originalFn, fnName)
 19 }
 20 
 21 /**
 22  * Wrap a function between 2 functions
 23  * 
 24  * @param {Function} originalFn the original function
 25  * @param {Function} beforeFn the function to call *before* the original function
 26  * @param {Function} afterFn the function to call *after* the original function
 27  * @returns {Function} The modified function
 28 */
 29 fnAttr.wrapCall	= function(originalFn, beforeFn, afterFn){
 30 	return function(){
 31 		var stopNow	= false;
 32 		// call beforeFn if needed
 33 		if( beforeFn )	stopNow = beforeFn(originalFn, arguments);
 34 		// forward the call to the original function
 35 		var result	= originalFn.apply(this, arguments);
 36 		// call afterFn if needed
 37 		if( afterFn )	afterFn(originalFn, arguments, result);
 38 		// return the result
 39 		return result;
 40 	}
 41 }

 47 var FnAttrClass	= function(originalFn, fnName){
 48 	this._currentFn	= originalFn;
 49 	this._fnName	= fnName	|| 'fnAttr'
 50 }
 51 
 52 FnAttrClass.prototype.end	=
 53 FnAttrClass.prototype.done	= function(){
 54 	return this._currentFn;
 55 }
 56 
 57 // export the class in node.js - if running in node.js
 58 if( typeof(window) === 'undefined' )	module.exports	= fnAttr;

 65 FnAttrClass.prototype.obsolete	= function(message){
 66 	var used	= false;
 67 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
 68 		if( used )	return;
 69 		used	= true;
 70 		console.warn(message || "Obsoleted function "+this._fnName+" called. Please update your code.");
 71 		console.trace();
 72 	}.bind(this));
 73 	return this;	// for chained API
 74 }

 76 FnAttrClass.prototype.timestamp	= function(){
 77 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
 78 		console.log("run at "+ new Date);
 79 	});
 80 	return this;	// for chained API
 81 };

 96 FnAttrClass.prototype.before	= function(beforeFn){
 97 	this._currentFn	= fnAttr.wrapCall(this._currentFn, beforeFn, null);
 98 	return this;	// for chained API
 99 };

101 FnAttrClass.prototype.after	= function(afterFn){
102 	this._currentFn	= fnAttr.wrapCall(this._currentFn, null, afterFn);
103 	return this;	// for chained API
104 };

116 FnAttrClass.prototype.time	= function(label){
117 	label	= label !== undefined ? label : this._fnName+".time()-"+Math.floor(Math.random()*9999).toString(36);
118 	this._currentFn	= fnAttr.wrapCall(this._currentFn, function(){
119 		console.time(label)
120 	}, function(){
121 		console.timeEnd(label)
122 	});
123 	return this;	// for chained API
124 };
