 23 /**
 24  * extract local methods - they are the ones allowed to access private field 
 25  * 
 26  * @param  {Function} klass the class constructor
 27  */
 28 PrivateForJS._extractLocalMethods	= function(klass){
 29 	var privateOkFns= [];
 30 	var protoNames	= Object.keys(klass.prototype);
 31 	for(var i = 0; i < protoNames.length; i++){
 32 		var protoName	= protoNames[i];
 33 		var fn		= klass.prototype[protoName];
 34 		if( typeof(fn) !== 'function' )	continue;
 35 		privateOkFns.push(fn);
 36 	}
 37 	return privateOkFns;
 38 };

 41 /**
 42  * define a private property on a given instance of a object class
 43  * @param  {Function} klass	the class of the intenciated object
 44  * @param  {Object} baseObject	the object instance
 45  * @param  {String} property	the property name
 46  * @return {undefined}		nothing
 47  */
 48 PrivateForJS.privateProperty	= function(klass, baseObject, property){
 49 // @TODO should i put a getter too ?
 50 // @TODO should i allow to update list of okfunctions ?
 51 // * - possible as it is stored in the class ctor itself, and not on the instance
 52 	QGetterSetter.defineGetter(baseObject, property, function aFunction(value, caller, property){
 53 		// generate privateOkFns if needed - functions which can access private properties
 54 		if( klass._PrivateForJS_OkFns === undefined ){
 55 			klass._PrivateForJS_OkFns	= PrivateForJS._extractLocalMethods(klass);
 56 		}
 57 		// check if caller is privateOk
 58 		var privateOkFns= klass._PrivateForJS_OkFns;
 59 		var isPrivateOk	= privateOkFns.indexOf(caller) !== -1;
 60 		if( isPrivateOk )	return value;
 61 		// get stackFrame and originId of the user
 62 		var stackFrame	= Stacktrace.parse()[2];
 63 		var originId	= stackFrame.fct + '@' + stackFrame.url + ':' + stackFrame.line;
 64 		// log the event
 65 		console.assert(false, 'access to private property', "'"+property+"'", 'from', originId);
 66 		// actually return the value
 67 		return value;
 68 	});
 69 };

 75 /**
 76  * Privatize 
 77  * @param  {[type]} klass      [description]
 78  * @param  {[type]} baseObject [description]
 79  * @return {[type]}            [description]
 80  */
 81 PrivateForJS.privatize	= function(klass, baseObject){
 82 	console.assert( baseObject.constructor == klass );
 83 	for(var property in baseObject){
 84 		var value	= baseObject[property]
 85 		if( property[0] !== '_' )		continue;
 86 		if(!baseObject.hasOwnProperty(property))continue;		
 87 		console.log('privatisation of', property)
 88 		PrivateForJS.privateProperty(klass, baseObject, property);
 89 	}
 90 }
