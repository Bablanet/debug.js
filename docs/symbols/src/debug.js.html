 25 /**
 26  * Display a message every time the function is called
 27  * 
 28  * @param {Function} fn the original function
 29  * @param {String} [message] message to display when the function is called. Optional
 30  * @returns {Function} The modified function
 31 */
 32 debug.obsolete	= function(originalFn, message){
 33 	return function(){
 34 		console.warn(message || "Obsoleted function called. Please update your code.");
 35 		console.trace();
 36 		// forward the call to the original function
 37 		return fn.apply(this, arguments);
 38 	}
 39 }

 41 /**
 42  * Wrap a function between 2 functions
 43  * 
 44  * @param {Function} originalFn the original function
 45  * @param {Function} beforeFn the function to call *before* the original function
 46  * @param {Function} afterFn the function to call *after* the original function
 47  * @returns {Function} The modified function
 48 */
 49 debug.wrapCall	= function(originalFn, beforeFn, afterFn){
 50 	return function(){
 51 		var stopNow	= false;
 52 		// call beforeFn if needed
 53 		if( beforeFn )	stopNow = beforeFn(originalFn, arguments);
 54 		// forward the call to the original function
 55 		var result	= originalFn.apply(this, arguments);
 56 		// call afterFn if needed
 57 		if( afterFn )	afterFn(originalFn, arguments, result);
 58 		// return the result
 59 		return result;
 60 	}
 61 }

 63 /**
 64  * Trigger the debugger when the function is called
 65  *
 66  * @param {Function} originalFn the original function
 67  * @param {Function} [conditionFn] this function should return true, when the breakpoint should be triggered. default to function(){ return true; }
 68  * @returns {Function} The modified function
 69 */
 70 debug.breakpoint	= function(fn, conditionFn){
 71 	conditionFn	= conditionFn	|| function(){ return true; };
 72 	return function(){
 73 		var stopNow	= conditionFn();
 74 		// if stopNow, trigger debugger
 75 		if( stopNow === true )	debugger;
 76 		// forward the call to the original function
 77 		return fn.apply(this, arguments);
 78 	}
 79 }

 81 /**
 82  * assert which actually try to stop the excecution
 83  * if debug.assert.useDebugger is falsy, throw an exception. else trigger the
 84  * debugger. It default to false
 85  * @param {Boolean} condition the condition which is asserted
 86  * @param {String} message the message which is display is condition is falsy
 87  * @param {?Boolean} useDebuggercondition the condition which is asserted
 88 */
 89 debug.assert	= function(condition, message, useDebugger){
 90 	if( condition )	return;
 91 	if( debug.assert.useDebugger || useDebugger )	debugger;
 92 	throw new Error(message	|| "assert Failed")
 93 }
 94 debug.assert.useDebugger	= false;

100 /**
101  * Ensure that the property is never NaN
102  * 
103  * @param {Object} baseObject the base object containing the property
104  * @param {String} property the property to check
105 */
106 debug.noNaN	= function(baseObject, property){
107 	baseObject.__defineQSetter__(property, function(value){
108 		console.assert(value === value, "property '"+property+"' is a NaN");
109 		return value;
110 	});
111 }

117  * @param {String} property the property to check
118  * @param {String} typeofString the exepected result of typeof(property)

143 /**
144  * Ensure the type of a variable using typeof or instanceof depending on klass
145  * 
146  * @param {Object} baseObject the base object containing the property
147  * @param {String} property the property to check
148  * @param {Object} klass the expected class
149 */
150 debug.checkType	= function(baseObject, property, klass){
151 	if( klass === Number ){
152 		debug.checkTypeof(baseObject, property, 'number');
153 	}else if( klass === String ){
154 		debug.checkTypeof(baseObject, property, 'string');
155 	}else{
156 		debug.checkInstanceof(baseObject, property, klass);		
157 	}
158 };

179 /**
180  * by default __defineGetter__ support only one function. Same for __defineSetter
181  * This is a annoying limitation. This little library declares 2 functions
182  * Object.__defineQGetter__ and Object.__defineQGetter__.
183  * They behave the same as their native sibling but support multiple functions.
184  * Those functions are called in the same order they got registered.
185  * 
186  * (I have no idea of the reasoning behind this limitation to one function. It seems
187  *  useless to me. This remind me of onclick of the DOM instead of a proper .addEventListener) 
188 */
189 (function(){
190 	var _QueuableGetterSetter	= function(baseObject, property){
191 		var _this	= this;
192 		this._getters	= [];
193 		this._setters	= [];
194 	
195 		var initialValue= baseObject[property];
196 		baseObject.__defineGetter__(property, function(){
197 			var value	= baseObject['__'+property];
198 			for(var i = 0; i < _this._getters.length; i++){
199 				value	= _this._getters[i](value)
200 			}
201 			return value;
202 		});
203 		baseObject.__defineSetter__(property, function(value){
204 			for(var i = 0; i < _this._setters.length; i++){
205 				value	= _this._setters[i](value)
206 			}
207 			baseObject['__'+property] = value;
208 		});
209 		// set the initialValue
210 		baseObject['__'+property]	= initialValue;
211 	};
212 	
213 	//////////////////////////////////////////////////////////////////////////////////
214 	// Override prototype of global ```Object```
215 	Object.prototype.__defineQGetter__	= function(property, getterFn){
216 		var name	= "__dbgGetSet_" + property;
217 		if( !this[name] ){
218 			this[name]	= new _QueuableGetterSetter(this, property);
219 		}
220 		this[name]._getters.push(getterFn)
221 	};
222 	
223 	Object.prototype.__defineQSetter__	= function(property, setterFn){
224 		var name	= "__dbgGetSet_" + property;
225 		if( !this[name] ){
226 			this[name]	= new _QueuableGetterSetter(this, property);
227 		}
228 		this[name]._setters.push(setterFn)
229 	};
230 })();
